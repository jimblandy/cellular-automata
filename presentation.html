                void
step(struct world *now, struct world *next) {
  int i, j;
  for (i = 1; i < HEIGHT - 1; i++)
    for (j = 1; j < WIDTH - 1; j++) {
      int neighbors = 0;
      int m, n;
      for (m = -1; m <= 1; m++)
        for (n = -1; n <= 1; n++)
          if (m || n)
            neighbors += now->cells[i + m][j + n];
      if (now->cells[i][j])
        next->cells[i][j] = (neighbors == 2 ||
                             neighbors == 3);
      else
        next->cells[i][j] = neighbors == 3;
    }
}
